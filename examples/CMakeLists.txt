function(add_r3d_example example_name source_file)
    add_executable(${example_name} ${source_file})
    target_compile_definitions(${example_name} PRIVATE R3D_ASSETS_PATH="${R3D_ROOT_PATH}/examples/assets/")
    target_link_libraries(${example_name} PRIVATE r3d)
    message(STATUS "Example '${example_name}' created with source file '${source_file}'")
endfunction()
